pipeline {
    agent any

    environment {
        IMAGE_NAME = 'todopy-app'
        CONTAINER_NAME = 'todopy-test'
    }

    stages {
        stage('Checkout') {
            steps {
                git url: 'https://github.com/chagasgb/todopy', branch: 'main'
            }
        }

        stage('build image') {
            steps {
                script {
                    sh "docker build -t $IMAGE_NAME ."
                }
            }
        }

        stage('pytest') {
            steps {
                script {
                    sh """
                        docker run --rm --name $CONTAINER_NAME $IMAGE_NAME pytest tests/test.py
                    """
                }
            }
        }
    }
}
